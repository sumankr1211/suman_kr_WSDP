#include <stdio.h>

int main(){
    int num1 , quo , rem ;

    printf("Enter number to be rounded off ..  ");
    scanf("%d",&num1);
    printf("Enter number whoose multiple is to be taken .. ");
    scanf("%d",&quo);

    rem = num1 %quo;

    if (rem < (quo-rem)){
        printf("number after rounding off = %d " , num1 - rem);
    }
    else {
        printf("number after rounding off = %d " , num1 - rem + quo );
    }
    
    return 0 ;
}
