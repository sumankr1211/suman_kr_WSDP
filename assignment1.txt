    printf("Enter number whoose multiple is to be taken .. ");

#include <stdio.h>

int main(){
    int r ;
    printf("Enter radius of cirlce ");
    scanf("%d",&r);

    int point = 0 ;
    int total = -1*r;
    for(; total <= r ; total++){
        int i = -1 * r ;

        while (i<=r){
            int res = total * total + i * i ;
            if (res < r){
                point ++;
            }
            i++;
        }
    }
    printf("total number of point in cirlce is %d ", point );
  
}
